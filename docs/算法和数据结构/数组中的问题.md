```js
function binraySearch(arr,target) {
  // 理解定义变量，边界定义
  	let left = 0, right = arr.length;
  	while(left < right) { // [left,right) , 不包括right
      const mid = parseInt((left+right)/2)
      const currItem = arr[mid]
      if(currItem === target) return mid;
      else if(currItem > target) {// [mid,right)
       left = mid + 1;
      } else if(currItem < target) { // [left,mid)
        right = mid;
      }
    }
  return -1;
}
```

### 题目1：移动非零元
> 空间时间复杂度，从0(n)优化到0(1); 从极端情况下优化
```js
/**
 * @param {number[]} nums
 * @return {void} Do not return anything, modify nums in-place instead.
 */
var moveZeroes = function(nums) {
    let k = 0;// 【0，k)中的元素为非0元素
    for(let i =0; i < nums.length; ++i) {
        if(nums[i]!==0) {
            // 处理自身和自身交换的极端情况
            if(i!==k) {
                // 交换nums[k]和nums[i]
                let temp = nums[k];
                nums[k] = nums[i];
                nums[i] = temp;
            }
            k++;
        }
    }
};
```

```js
/**
 * @param {number[]} nums
 * @param {number} val
 * @return {number}
 */
var removeElement = function(nums, val) {
    let k = 0; // [0,k)确保该区间为非val元素，
    // k其实这里充当的角色就是非val组成的数组的长度
    for(let i =0; i < nums.length;++i) {
        if(nums[i]!== val) {
                nums[k] = nums[i];
                k++;
        } 
    }
    return k;
};
```

### 26. 删除有序数组中的重复项
> 重点是：数组中的元素有序排列
```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var removeDuplicates = function(nums) {
    const len = nums.length;
    // 0 1个元素都不可能重复
    if(len < 2) return len;

    // 从第二个元素开始
    let fast = 1,slow = 1;
    while(fast < len) {
        if(nums[fast] !== nums[fast-1]){
            nums[slow] = nums[fast];
            slow++;
        }
        fast++;
    }
    return slow;

};
```
